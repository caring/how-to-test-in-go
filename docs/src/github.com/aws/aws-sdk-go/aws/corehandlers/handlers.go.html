// signature doesn't expire before it is sent. This can happen when a request
// is built and signed significantly before it is sent. Or significant delays

	Name: "core.ValidateReqSigHandler",
	Fn: func( *request.Request) {
		// Unsigned requests are not signed

		}

		 := .Time

		// Would be improved with aws/aws-sdk-go#423
		if .Add(5 * time.Minute).After(time.Now()) {
			return
		}

		fmt.Println("request expired, resigning")
		.Sign()
	},
}

func ( *request.Request,  error) {
	// Prevent leaking if an HTTPResponse was returned. Clean up
	// the body.
	if .HTTPResponse != nil {
		.HTTPResponse.Body.Close()
	}
	// Capture the case where url.Error is returned for error processing
	// response. e.g. 301 without location header comes back as string
	// error and r.HTTPResponse is nil. Other URL redirect errors will
	// comeback in a similar method.
	if ,  := .(*url.Error);  && .Err != nil {
		if  := reStatusCode.FindStringSubmatch(.Err.Error());  != nil {
			,  := strconv.ParseInt([1], 10, 64)
			.HTTPResponse = &http.Response{
				StatusCode: int(),
				Status:     http.StatusText(int()),
				Body:       ioutil.NopCloser(bytes.NewReader([]byte{})),
			}
			return
		}
	}
	if .HTTPResponse == nil {
		// Add a dummy request response object to ensure the HTTPResponse
		// value is consistent.
		.HTTPResponse = &http.Response{
			StatusCode: int(0),
			Status:     http.StatusText(int(0)),
			Body:       ioutil.NopCloser(bytes.NewReader([]byte{})),
		}
	}
	// Catch all request errors, and let the default retrier determine
	// if the error is retryable.
	.Error = awserr.New(request.ErrCodeRequestError, "send request failed", )

	// Override the error with a context canceled error, if that was canceled.
	 := .Context()
	select {
	case <-.Done():
		.Error = awserr.New(request.CanceledErrorCode,
			"request context canceled", .Err())
		.Retryable = aws.Bool(false)
	default:
	}
}

// AfterRetryHandler performs final checks to determine if the request should
// be retried and how long to delay.
var AfterRetryHandler = request.NamedHandler{
	Name: "core.AfterRetryHandler",
	Fn: func( *request.Request) {
		// If one of the other handlers already set the retry state
		// we don't want to override it based on the service's state
		if .Retryable == nil || aws.BoolValue(.Config.EnforceShouldRetryCheck) {
			.Retryable = aws.Bool(.ShouldRetry())
		}

		if .WillRetry() {
			.RetryDelay = .RetryRules()

			if  := .Config.SleepDelay;  != nil {
				// Support SleepDelay for backwards compatibility and testing
				(.RetryDelay)
			} else if  := aws.SleepWithContext(.Context(), .RetryDelay);  != nil {
				.Error = awserr.New(request.CanceledErrorCode,
					"request context canceled", )
				.Retryable = aws.Bool(false)
				return
			}

			// when the expired token exception occurs the credentials
			// need to be expired locally so that the next request to
			// get credentials will trigger a credentials refresh.
			if .IsErrorExpired() {
				.Config.Credentials.Expire()
			}

			.RetryCount++
			.Error = nil
		}
	}}

// ValidateEndpointHandler is a request handler to validate a request had the
// appropriate Region and Endpoint set. Will set r.Error if the endpoint or
// region is not valid.
var ValidateEndpointHandler = request.NamedHandler{Name: "core.ValidateEndpointHandler", Fn: func( *request.Request) {
	if .ClientInfo.SigningRegion == "" && aws.StringValue(.Config.Region) == "" {
		.Error = aws.ErrMissingRegion
	} else if .ClientInfo.Endpoint == "" {
		.Error = aws.ErrMissingEndpoint
	}
