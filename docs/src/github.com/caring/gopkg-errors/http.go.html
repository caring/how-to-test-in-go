// HTTPFromGrpc converts a gRPC error code to an HTTP code. See:
// https://grpc.github.io/grpc/core/md_doc_statuscodes.html
// https://developer.mozilla.org/en-US/docs/Web/HTTP/Status
func ( codes.Code) int {

	// Use a lookup table to convert a gRPC status code to a HTTP status code
	 := [17]int{
		http.StatusOK,                           // gRPC 0 (OK) -> HTTP 200
		http.StatusGone,                         // gRPC 1 (Cancelled) -> HTTP 410
		http.StatusUnprocessableEntity,          // gRPC 2 (Unknown) -> HTTP 422
		http.StatusBadRequest,                   // gRPC 3 (Invalid Argument) -> HTTP 400
		http.StatusRequestTimeout,               // gRPC 4 (Deadline Exceeded) -> HTTP 408
		http.StatusNotFound,                     // gRPC 5 (Not Found) -> HTTP 404
		http.StatusConflict,                     // gRPC 6 (Already Exists) -> HTTP 409
		http.StatusForbidden,                    // gRPC 7 (Permission Denied) -> HTTP 403
		http.StatusInternalServerError,          // gRPC 8 (Resource Exhausted) -> HTTP 500
		http.StatusPreconditionFailed,           // gRPC 9 (Failed Precondition) -> HTTP 412
		http.StatusResetContent,                 // gRPC 10 (Aborted) -> HTTP 205
		http.StatusRequestedRangeNotSatisfiable, // gRPC 11 (Out Of Range) -> HTTP 416
		http.StatusNotImplemented,               // gRPC 12 (Unimplemented) -> HTTP 501
		http.StatusInternalServerError,          // gRPC 13 (Internal) -> HTTP 500
		http.StatusServiceUnavailable,           //gRPC 14 (Unavailable) -> HTTP 503
		http.StatusInternalServerError,          // gRPC 15 (Data loss) -> HTTP 500
		http.StatusUnauthorized,                 // gRPC 16 (Unauthenticated) -> HTTP 401
	}

	if  > 17 {
		return http.StatusInternalServerError
	}

	return []
}

// ToHTTP writes the error to the http response.
func ( *status.Status,  http.ResponseWriter) error {
	.Header().Set("Content-Type", "application/json")
	.WriteHeader(HTTPFromGrpc(.Code()))
	return json.NewEncoder().Encode(.Code())
}
