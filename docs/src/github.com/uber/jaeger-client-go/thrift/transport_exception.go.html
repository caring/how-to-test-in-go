// Thrift Transport exception
type TTransportException interface {
	TException
	TypeId() int
	Err() error
}

const (
	UNKNOWN_TRANSPORT_EXCEPTION = 0
	NOT_OPEN                    = 1
	ALREADY_OPEN                = 2
	TIMED_OUT                   = 3
	END_OF_FILE                 = 4
)

type tTransportException struct {
	typeId int
	err    error
	msg    string
}

var _ TTransportException = (*tTransportException)(nil)

func (tTransportException) () TExceptionType {
	return TExceptionTypeTransport
}

func ( *tTransportException) () int {
	return .typeId
}

func ( *tTransportException) () string {
	return .msg
}

func ( *tTransportException) () error {
	return .err
}

func ( *tTransportException) () error {
	return .err
}

func ( *tTransportException) () bool {
	return .typeId == TIMED_OUT
}
