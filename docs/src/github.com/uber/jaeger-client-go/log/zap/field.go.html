// Trace creates a field that extracts tracing information from a context and
// includes it under the "trace" key.
//
// Because the opentracing APIs don't expose this information, the returned
// zap.Field is a no-op for contexts that don't contain a span or contain a
// non-Jaeger span.
// TODO: delegate to `spanContext`
func ( context.Context) zapcore.Field {
	if  == nil {
		return zap.Skip()
	}
	return zap.Object("trace", trace{})
}

type trace struct {
	ctx context.Context
}

func ( trace) ( zapcore.ObjectEncoder) error {
	 := opentracing.SpanFromContext(.ctx)
	if  == nil {
		return nil
	}
	,  := .Context().(jaeger.SpanContext)
	if ! {
		return nil
	}
	if !.IsValid() {
		return fmt.Errorf("invalid span: %v", .SpanID())
	}
	.AddString("span", .SpanID().String())
	.AddString("trace", .TraceID().String())
	.AddBool("sampled", .IsSampled())
	return nil
}

func ( spanContext) ( zapcore.ObjectEncoder) error {
	 := jaeger.SpanContext()
	.AddString("trace", .TraceID().String())
	.AddString("span", .SpanID().String())
	.AddString("parent", .ParentID().String())
	.AddBool("debug", .IsDebug())
	.AddBool("sampled", .IsSampled())
	.AddBool("firehose", .IsFirehose())
	.AddArray("baggage", baggageKVs())
	return nil
}

type reference opentracing.SpanReference

func ( reference) ( zapcore.ObjectEncoder) error {
	if .Type == opentracing.ChildOfRef {
		.AddString("type", "child_of")
	} else if .Type == opentracing.FollowsFromRef {
		.AddString("type", "follows_from")
	} else {
		.AddString("type", fmt.Sprintf("unknown(%d)", .Type))
	}

	if ,  := .ReferencedContext.(jaeger.SpanContext);  {
		.AddString("span", .SpanID().String())
		.AddString("trace", .TraceID().String())
	}

	return nil
}

type span struct {
	span *jaeger.Span
}

// Span creates a zap.Field that marshals all information contained in a jaeger span
func ( opentracing.Span) zapcore.Field {
	if  == nil {
		return zap.String("err", "nil span")
	}
	if ,  := .(*jaeger.Span);  {
		return zap.Object("span", span{span: })
	}
	return zap.String("err", "non jaeger span")
}

func ( span) ( zapcore.ObjectEncoder) error {
	.AddObject("context", spanContext(.span.SpanContext()))
	.AddString("operation_name", .span.OperationName())
	.AddDuration("duration", .span.Duration())
	.AddTime("start_time", .span.StartTime())

	.AddArray("logs", logRecords(.span.Logs()))
	.AddObject("tags", tags(.span.Tags()))
	.AddArray("span_refs", references(.span.References()))
	return nil
}
